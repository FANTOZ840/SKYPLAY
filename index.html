let multiplier = 1.0; let isRunning = false; function updateMultiplier() { if (isRunning) { multiplier += 0.1; // Example increment logic document.getElementById('multiplierDisplay').innerText = multiplier.toFixed(2) + 'x'; setTimeout(updateMultiplier, 100); // Update every 100ms } } function startGame() { isRunning = true; updateMultiplier(); } // Call startGame() when the game starts
