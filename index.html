let activeBets = { 1: {placed:false, amount:100}, 2: {placed:false, amount:50}, 3: {placed:false, amount:20} }; function changeBet(box, value){ let input = document.getElementById("bet"+box); input.value = Math.max(10, parseInt(input.value) + value); } function placeBet(box){ let amount = parseInt(document.getElementById("bet"+box).value); activeBets[box].placed = true; activeBets[box].amount = amount; document.getElementById("result"+box).innerText = " Bet placed: Sh."+amount; }
